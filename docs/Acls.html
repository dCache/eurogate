The EuroStore Acl's

User Administration

CommandAcl
create user <user> user.*.create
create group <group> user.*.create
destroy principal <principal> user.*.create
add <principal> to <group> user.<group>.access
set <principal> [<key>=<value> ...] user.<principal>.access
show groups -
show group <group> -
show principal <principal> -
set passwd user.<principal>.setpassword

Acl Administration

CommandAcl
create acl <aclName> super.access
destroy acl <aclName> super.access
show acl <aclName> -
add access [-allowed|-denied] <acl> <principal> acl.<acl>.access

Pvl Administration

CommandAcl
create pvr <pvrName> pvr.*.create
create drive -pvr=<pvrName> <driveName> pvr.<pvrName>.modify
set drive -pvr=<pvrName> <driveName> ... pvr.<pvrName>.modify
create cartridgeDescriptor <name> pvl.*.modify
create volumeDescriptor <name> pvl.*.modify
